#include <iostream>
using namespace std;

void swap(int& n, int& m)
{
	n=n+m;
	m=n-m;
	n=n-m;
}
int main()
{
    int num1, num2;
    cout<<"Input 2 numbers n1 and n2: ";
	cin>>num1>>num2;
	cout<<endl;
    cout<<"Swapped value of n1 and n2: ";
	swap (num1, num2);

}
